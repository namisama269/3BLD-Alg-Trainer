Rendering Only (visualCube.js)

// Just need visualCube.js for static display
const canvas = document.getElementById('myCube');
const ctx = canvas.getContext('2d');

const vc = new VisualCube(
    canvas.width,   // width
    canvas.height,  // height
    180,            // scale
    -0.4,           // thetaX (radians)
    -0.6,           // thetaY (radians)
    0,              // thetaZ (radians)
    3,              // cubeSize
    0.08,           // gapSize
    0.65            // edgeGapRatio
);

vc.showBaseColor = true;
vc.baseColor = "#000000";
vc.stickerBorderColor = null;

// 54-char string: U(9) + R(9) + F(9) + D(9) + L(9) + B(9)
vc.cubeString = "UUUUUUUUURRRRRRRRFFFFFFFFFDDDDDDDDDLLLLLLLLLBBBBBBBBB";

vc.drawCube(ctx);

With Cube Manipulation (cube.js)

// Include: cube.js + visualCube.js

// Create cube using cube.js
const cube = new Cube();

// Apply moves (standard notation)
cube.move("R U R' U'");

// Get state as 54-char string for rendering
const cubeString = cube.asString();

// Render
const vc = new VisualCube(600, 600, 180, -0.4, -0.6, 0, 3, 0.08, 0.65);
vc.cubeString = cubeString;
vc.drawCube(ctx);

// Reset to solved
cube.identity();

// Check if solved
if (cube.isSolved()) {
    console.log("Solved!");
}

// Random scramble
Cube.initSolver();  // Call once to initialize
const scramble = Cube.scramble();
cube.move(scramble);
